156 void Object_3D::set_instance_data(Renderer* handler, vector<T> info)
157 {
158  handler->busy_queue.lock();
159  glBindVertexArray(VAO);

162  glBufferData(GL_SHADER_STORAGE_BUFFER, info.size()*sizeof(T),

164  handler->busy_queue.unlock();
